QUESTION:
Create a React app that has two pages: Home and About. The Home page should display a welcome message and a link to the about page. The About page should display information about your app.
ANSWER:

Install 'react-router-dom':
npm install react-router-dom

App.js
// App.js
import React from 'react';
import { BrowserRouter as Router, Route, Link, Switch } from 'react-router-dom';
import Home from './Home';
import About from './About';

const App = () => {
  return (
    <Router>
      <div>
        <nav>
          <ul>
            <li>
              <Link to="/">Home</Link>
            </li>
            <li>
              <Link to="/about">About</Link>
            </li>
          </ul>
        </nav>

        <Switch>
          <Route path="/about">
            <About />
          </Route>
          <Route path="/">
            <Home />
          </Route>
        </Switch>
      </div>
    </Router>
  );
};

export default App;


Home.js
// Home.js
import React from 'react';
import { Link } from 'react-router-dom';

const Home = () => {
  return (
    <div>
      <h2>Welcome to the Home page!</h2>
      <p>
        Click <Link to="/about">here</Link> to visit the About page.
      </p>
    </div>
  );
};

export default Home;

About.js
// About.js
import React from 'react';

const About = () => {
  return (
    <div>
      <h2>About This App</h2>
      <p>This is a simple React app with two pages: Home and About.</p>
    </div>
  );
};

export default About;
